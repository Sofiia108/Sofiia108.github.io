function getMaxDigit(number){
    return Math.max(...Array.from(String(number), Number));
}

var exponent = function(a, n) 
{
   if (n === 0) { return 1}
   else{ 
       return a * exponent(a, n-1);
   }
};

function getSalaryAfterTax(tax1, tax2, salaryWithoutTax){
    return (salaryWithoutTax * (1-(tax1+tax2/100))).toFixed(2);
}

function convertCurrency(amountOfMoney){

    if(amountOfMoney.includes("$")){
        return (parseFloat(amountOfMoney)*25).toLocaleString('en-US', {
            style: 'currency',
            currency: 'UAH',
          });
    }

    if(amountOfMoney.includes("UAH")){
        return (parseFloat(amountOfMoney)/25).toLocaleString('en-US', {
            style: 'currency',
            currency: 'USD',
          });
    }
}

convertCurrency("2500UAH");

function deleteDuplicateLetter(str){
    {
        var unique = '';
        for (var i = 0; i < str.length; i++) {
          if (str.lastIndexOf(str[i]) == str.indexOf(str[i])) {
            unique += str[i];
          }
        }
        return unique;
}}

deleteDuplicateLetter("pleeease");
